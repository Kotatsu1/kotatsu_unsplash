CREATE TABLE collections (
  collection_id serial PRIMARY KEY,
  title TEXT NOT NULL,
  description TEXT,
  fk_user_id text REFERENCES "User"(id) NOT NULL
);


CREATE TABLE collection_content (
  content_id serial PRIMARY KEY,
  public_id TEXT NOT NULL,
  fk_collection_id integer REFERENCES collections(collection_id) NOT NULL
);
